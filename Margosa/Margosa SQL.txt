CREATE TABLE Family (
    FID INT GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
    FAMILYNAME VARCHAR(255) NOT NULL,
    FAMILYPIN VARCHAR(6) NOT NULL,
    HID INT
);

CREATE TABLE House 
(
    	HID INT GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
   	HOUSENUM VARCHAR(255) NOT NULL,
    	HOUSESTREET VARCHAR(255) NOT NULL,
	FID INT,
    	FOREIGN KEY (FID) REFERENCES Family (FID)
);

CREATE TABLE Resident
(
    RID INT GENERATED ALWAYS AS IDENTITY(START WITH 1, INCREMENT BY 1) PRIMARY KEY,
    FIRSTNAME VARCHAR(255) NOT NULL,
    LASTNAME VARCHAR(255) NOT NULL,
    GENDER VARCHAR(255) NOT NULL,
    BIRTHDAY DATE,
    EMAIL VARCHAR(255) NOT NULL,
    PHONENUM VARCHAR(255) NOT NULL,
    USERNAME VARCHAR(255) NOT NULL,
    PASSWORD VARCHAR(255) NOT NULL,
    APPROVED BOOLEAN DEFAULT false,
    FID INT,
    HID INT,
    FOREIGN KEY (FID) REFERENCES Family (FID),
    FOREIGN KEY (HID) REFERENCES House (HID)
);


CREATE TABLE Car 
(
    	CID INT GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
    	CARBRAND VARCHAR(255) NOT NULL,
    	CARMODEL VARCHAR(255) NOT NULL,
    	CARCOLOR VARCHAR(255) NOT NULL,
    	CARPLATENUM VARCHAR(255) NOT NULL,
    	CARTYPE VARCHAR(255) NOT NULL,
	FID INT,
    	FOREIGN KEY (FID) REFERENCES Family (FID)
);


CREATE TABLE Visitor
(
    VID INT GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
    VISITORNAME VARCHAR(255) NOT NULL,
    VISITORPHONENUM VARCHAR(255) NOT NULL,
    VISITORVEHICLENUM VARCHAR(255) NOT NULL,
    VISITORENTRYDATE DATE,
    VISITORENTRYTIME TIME,
    VISITOREXITDATE DATE,
    VISITOREXITTIME TIME,
    VISITORADDEDDATE DATE,
    VISITORADDEDTIME TIME,
    HOUSENUM VARCHAR(255),
    HOUSESTREET VARCHAR(255),
    FID INT,
    FOREIGN KEY (FID) REFERENCES Family (FID)
);


CREATE TABLE Admin
(
	AID INT GENERATED ALWAYS AS IDENTITY(Start with 1, Increment by 1) PRIMARY KEY, 
	USERNAME VARCHAR(255) NOT NULL, 
	PASSWORD VARCHAR(255) NOT NULL, 
	NAME VARCHAR(255) NOT NULL
);
